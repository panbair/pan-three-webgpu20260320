# 打字训练游戏 - 三模式统一架构设计

## 核心设计思路

采用**共享核心 + 独立模式**的架构，避免代码重复，便于维护和扩展。

#### 模式A: 陨石防御 (modes/meteor-defense.ts)

```typescript
export class MeteorDefenseMode extends GameMode {
  readonly name = '陨石防御'
  readonly description = '陨石从天而降，输入字母摧毁它们！'

  private meteors: Meteor[] = []
  private spawnTimer = 0
  private spawnInterval = 2000

  async onStart() {
    // 创建防护墙
    const wall = this.createGroundWall()
    this.engine.scene.add(wall)

    // 创建玩家飞船
    const ship = this.createPlayerShip()
    this.engine.scene.add(ship)
  }

  async onUpdate(deltaTime: number) {
    // 生成陨石
    this.spawnTimer += deltaTime
    if (this.spawnTimer > this.spawnInterval) {
      this.spawnMeteor()
      this.spawnTimer = 0
    }

    // 更新陨石（重力下落）
    this.meteors.forEach(m => {
      m.mesh.position.y -= m.speed * deltaTime
    })

    // 检测碰撞（陨石撞击地面）
    this.checkGroundCollision()
  }

  private spawnMeteor() {
    // 创建陨石 + 3D字母
    const meteor = {
      mesh: new THREE.Mesh(meteorGeometry, meteorMaterial),
      letter: this.engine.text3D.create(randomLetter()),
      speed: 2 + Math.random() * 2
    }
    meteor.mesh.position.set(
      (Math.random() - 0.5) * 30,
      20,
      0
    )
    this.meteors.push(meteor)
    this.engine.scene.add(meteor.mesh)
  }
}
```

#### 模式B: 星球保护 (modes/planet-protection.ts)

```typescript
export class PlanetProtectionMode extends GameMode {
  readonly name = '星球保护'
  readonly description = '敌人从四面八方进攻，保护中央星球！'

  private planet: THREE.Mesh
  private enemies: Enemy[] = []

  async onStart() {
    // 创建中央星球
    this.planet = this.createPlanet()
    this.planet.position.set(0, 0, 0)
    this.engine.scene.add(this.planet)

    // 创建星空背景
    this.createStarfield()
  }

  async onUpdate(deltaTime: number) {
    // 敌人向星球移动（环形轨道）
    this.enemies.forEach(enemy => {
      // 使用物理引擎计算轨道运动
      const direction = this.planet.position.clone().sub(enemy.mesh.position).normalize()
      enemy.mesh.position.add(direction.multiplyScalar(enemy.speed * deltaTime))
    })

    // 检测敌人碰撞（撞击星球）
    this.checkPlanetCollision()
  }

  private spawnEnemy() {
    // 敌人从随机方向生成
    const angle = Math.random() * Math.PI * 2
    const distance = 50
    const enemy = {
      mesh: this.createEnemyMesh(),
      letter: this.engine.text3D.create(randomLetter()),
      position: new THREE.Vector3(
        Math.cos(angle) * distance,
        Math.sin(angle) * distance,
        0
      )
    }
    this.enemies.push(enemy)
    this.engine.scene.add(enemy.mesh)
  }
}
```

#### 模式C: 文字冲浪 (modes/tunnel-rush.ts)

```typescript
export class TunnelRushMode extends GameMode {
  readonly name = '文字冲浪'
  readonly description = '高速隧道中，输入字母获得速度！'

  private tunnel: THREE.Mesh
  private blocks: Block[] = []
  private speed: number = 30

  async onStart() {
    // 创建赛博朋克隧道
    this.tunnel = this.createCyberTunnel()
    this.engine.scene.add(this.tunnel)

    // 设置相机跟随模式
    this.engine.controls.autoRotate = false
  }

  async onUpdate(deltaTime: number) {
    // 字符块高速向玩家移动
    this.blocks.forEach(block => {
      block.mesh.position.z += this.speed * deltaTime

      // 连击加速
      if (block.isHit) {
        this.speed += 0.1
      }
    })

    // 移除超出视野的块
    this.removePassedBlocks()

    // 生成新块
    this.spawnBlock()
  }

  private createCyberTunnel() {
    // 赛博朋克风格隧道（发光、霓虹）
    const geometry = new THREE.CylinderGeometry(10, 10, 200, 32, 1, true)
    const material = new THREE.MeshBasicMaterial({
      color: 0x00ff00,
      wireframe: true,
      side: THREE.BackSide
    })
    return new THREE.Mesh(geometry, material)
  }

  private spawnBlock() {
    // 在隧道尽头生成字符块
    const block = {
      mesh: this.createBlockMesh(),
      letter: this.engine.text3D.create(randomLetter()),
      position: new THREE.Vector3(
        (Math.random() - 0.5) * 10,
        (Math.random() - 0.5) * 10,
        -100
      )
    }
    this.blocks.push(block)
    this.engine.scene.add(block.mesh)
  }
}
```

## 模式管理器 (modes/index.ts)

```typescript
import { GameMode } from './base'
import { MeteorDefenseMode } from './meteor-defense'
import { PlanetProtectionMode } from './planet-protection'
import { TunnelRushMode } from './tunnel-rush'

export class ModeManager {
  private engine: GameEngine
  private currentMode: GameMode | null = null
  private modes: Map<string, GameMode>

  constructor(engine: GameEngine) {
    this.engine = engine
    this.modes = new Map([
      ['meteor', new MeteorDefenseMode(engine)],
      ['planet', new PlanetProtectionMode(engine)],
      ['tunnel', new TunnelRushMode(engine)]
    ])
  }

  async switchMode(modeKey: string) {
    // 停止当前模式
    if (this.currentMode) {
      await this.currentMode.onStop()
    }

    // 切换新模式
    this.currentMode = this.modes.get(modeKey)!
    await this.currentMode.onStart()
  }

  async update(deltaTime: number) {
    if (this.currentMode) {
      await this.currentMode.onUpdate(deltaTime)
    }
  }

  getModeInfo() {
    return Array.from(this.modes.entries()).map(([key, mode]) => ({
      key,
      name: mode.name,
      description: mode.description
    }))
  }
}
```

## 游戏主入口 (index.vue)

```vue
<template>
  <div class="game-container">
    <!-- 3D Canvas -->
    <div ref="canvasContainer" class="canvas-container"></div>

    <!-- 模式选择界面 -->
    <ModeSelector
      v-if="gameState === 'select'"
      :modes="modeManager.getModeInfo()"
      @select="startGame"
    />

    <!-- 游戏内HUD -->
    <HUD
      v-if="gameState === 'playing'"
      :stats="gameStats"
      :mode="currentModeName"
      @pause="pauseGame"
    />

    <!-- 游戏结束界面 -->
    <GameOver
      v-if="gameState === 'gameover'"
      :stats="gameStats"
      :mode="currentModeName"
      @restart="restartGame"
      @back="backToSelect"
    />
  </div>
</template>

<script setup lang="ts">
import { ref, onMounted, onBeforeUnmount } from 'vue'
import { GameEngine } from './core/engine'
import { ModeManager } from './modes'
import ModeSelector from './ui/ModeSelector.vue'
import HUD from './ui/HUD.vue'
import GameOver from './ui/GameOver.vue'

const canvasContainer = ref<HTMLDivElement>()
const engine = ref<GameEngine>()
const modeManager = ref<ModeManager>()
const gameState = ref<'select' | 'playing' | 'paused' | 'gameover'>('select')
const gameStats = ref<GameStats>({} as GameStats)
const currentModeName = ref('')

let animationId: number = 0

onMounted(async () => {
  // 初始化引擎
  engine.value = new GameEngine()
  await engine.value.init()
  canvasContainer.value!.appendChild(engine.value.renderer.domElement)

  // 初始化模式管理器
  modeManager.value = new ModeManager(engine.value)

  // 启动游戏循环
  startGameLoop()
})

const startGame = async (modeKey: string) => {
  gameState.value = 'playing'
  currentModeName.value = modeManager.value.getModeInfo().find(m => m.key === modeKey)!.name

  // 重置统计
  gameStats.value = {
    score: 0,
    wpm: 0,
    combo: 0,
    lives: 3,
    timeElapsed: 0
  }

  // 切换模式
  await modeManager.value.switchMode(modeKey)
}

const startGameLoop = () => {
  const loop = (time: number) => {
    const deltaTime = (time - lastTime) / 1000
    lastTime = time

    // 更新物理世界
    engine.value?.world.step()

    // 更新模式
    if (gameState.value === 'playing') {
      modeManager.value?.update(deltaTime)
    }

    // 渲染
    engine.value?.renderer.render(engine.value.scene, engine.value.camera)

    animationId = requestAnimationFrame(loop)
  }

  let lastTime = performance.now()
  animationId = requestAnimationFrame(loop)
}

// 键盘输入处理
const handleKeyPress = (event: KeyboardEvent) => {
  if (gameState.value !== 'playing') return

  // 传递给当前模式处理
  const char = event.key.toUpperCase()
  modeManager.value?.handleInput(char)
}

window.addEventListener('keypress', handleKeyPress)
</script>
```

### 1. 对象池 (core/objects/pool.ts)

```typescript
export class ObjectPool<T> {
  private pool: T[] = []
  private factory: () => T
  private maxSize: number

  constructor(factory: () => T, maxSize: number = 100) {
    this.factory = factory
    this.maxSize = maxSize
  }

  get(): T {
    return this.pool.pop() || this.factory()
  }

  release(obj: T) {
    if (this.pool.length < this.maxSize) {
      this.pool.push(obj)
    }
  }

  clear() {
    this.pool = []
  }
}

// 使用示例
const meteorPool = new ObjectPool(() => createMeteor(), 50)
const meteor = meteorPool.get()
// ... 使用meteor ...
meteorPool.release(meteor)
```

### 2. 爆炸特效 (core/objects/explosion.ts)

```typescript
export class ExplosionSystem {
  private scene: THREE.Scene
  private particles: THREE.Mesh[] = []

  constructor(scene: THREE.Scene) {
    this.scene = scene
  }

  create(position: THREE.Vector3, color: number = 0xff0000) {
    const particleCount = 20
    const geometry = new THREE.SphereGeometry(0.2)
    const material = new THREE.MeshBasicMaterial({ color })

    for (let i = 0; i < particleCount; i++) {
      const particle = new THREE.Mesh(geometry, material.clone())
      particle.position.copy(position)

      // 随机速度
      const velocity = new THREE.Vector3(
        (Math.random() - 0.5) * 10,
        (Math.random() - 0.5) * 10,
        (Math.random() - 0.5) * 10
      )

      this.scene.add(particle)
      this.particles.push({ mesh: particle, velocity })

      // GSAP动画
      gsap.to(particle.position, {
        x: position.x + velocity.x,
        y: position.y + velocity.y,
        z: position.z + velocity.z,
        duration: 1,
        ease: 'power2.out'
      })
      gsap.to(material, {
        opacity: 0,
        duration: 1,
        onComplete: () => {
          this.scene.remove(particle)
        }
      })
    }
  }

  clear() {
    this.particles.forEach(p => this.scene.remove(p.mesh))
    this.particles = []
  }
}
```

### 2. 批量渲染

```typescript
// 使用 InstancedMesh 渲染大量相同物体
export class InstancedRenderer {
  private instancedMesh: THREE.InstancedMesh
  private count = 0
  private matrix = new THREE.Matrix4()

  constructor(maxCount: number, geometry: THREE.BufferGeometry, material: THREE.Material) {
    this.instancedMesh = new THREE.InstancedMesh(geometry, material, maxCount)
  }

  addInstance(position: THREE.Vector3, scale: number = 1) {
    this.matrix.makeTranslation(position.x, position.y, position.z)
    this.matrix.scale(new THREE.Vector3(scale, scale, scale))
    this.instancedMesh.setMatrixAt(this.count++, this.matrix)
    this.instancedMesh.instanceMatrix.needsUpdate = true
  }

  reset() {
    this.count = 0
  }
}
```

## 总结

**三种模式完全可行！** 核心优势：

✅ **架构统一**：共享核心代码，避免重复
✅ **易于扩展**：新增模式只需继承基类
✅ **性能优秀**：对象池、批量渲染、资源共享
✅ **用户体验**：模式选择、统一统计、排行榜
✅ **可维护性**：清晰的模块划分，便于后续开发

推荐开发顺序：**模式A → 模式B → 模式C**，每个模式完成后即可测试，逐步完善。
